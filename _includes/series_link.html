{%- comment %} series_link.html Generates a link to the specific series section on the corresponding author's page. Parameters: - series: The title of the series (required). - link_text: Optional override for display text. Defaults to the series title. {% endcomment -%} {%- assign series_title = include.series | strip -%} {%- assign link_text = include.link_text | default: series_title | strip -%} {%- assign final_url = "" -%} {%- assign author_name = "" -%} {%- if series_title != "" -%} {%- comment %} Find *a* book belonging to this series to determine the author. We assume all books in a series have the same author listed consistently in the 'book_author' front matter variable, and that this name matches the 'title' of the author's page. {% endcomment -%} {%- assign books_in_series = site.books | where: "series", series_title -%} {%- if books_in_series.size > 0 -%} {%- assign book = books_in_series | first -%} {%- assign author_name = book.book_author -%} {%- endif -%} {%- if author_name != "" -%} {%- comment %} Found an author, now find the URL for that author's page {% endcomment -%} {%- capture author_url_capture -%} {% include get_url_from_title.html title=author_name layout="author_page" %} {%- endcapture -%} {%- assign author_page_url = author_url_capture | strip -%} {%- if author_page_url != "" -%} {%- comment %} Successfully found the author page URL. Generate the anchor slug. {% endcomment -%} {%- assign series_slug = series_title | slugify -%} {%- comment %} Construct the final URL: author_page_url#series_slug {% endcomment -%} {%- assign final_url = author_page_url | append: "#" | append: series_slug -%} {%- endif -%} {%- endif -%} {%- endif -%} {%- comment %} Output the link if we constructed a valid URL, otherwise just display the text. {% endcomment -%} {%- if final_url != "" -%} {{ link_text }} {%- else -%} {%- comment %} Fallback: Show text if series, author, or author page wasn't found {% endcomment -%} {{ link_text }} {%- comment %} Optional: Add HTML comment for debugging build failures {% endcomment %} {%- endif -%}
