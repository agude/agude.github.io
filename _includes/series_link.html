{%- comment %} The HTML is *VERY* sensitive to new lines, so we fill with {{- null -}} which does nothing _except_ consume a new line on eitherside. Otherwise it doens't work inside footnotes. See: https://stackoverflow.com/questions/50467557/jekyll-on-github-pages-how-to-include-external-content-inside-footnotes {% endcomment -%} {%- assign target_series = include.series | replace: newline,' ' | downcase | strip -%} {%- comment -%}Assign link_text here incase we don't find a link. {%- endcomment -%} {%- assign link_text = include.seires | replace: newline, ' ' | strip -%} {%- assign url = null -%} {%- comment %}Try to find the series and set a URL{% endcomment -%} {%- for page in site.pages -%} {%- if page.series_name == include.series and page.layout == "series_page" -%} {%- assign url = page.url -%} {%- assign link_text = page.series_name | replace: newline, ' ' | strip -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- comment %}Wrap the name in tags.{% endcomment -%} {%- capture wrapped_name -%}{{ link_text }}{%- endcapture -%} {%- comment %}Wrap again with a link if we find one.{% endcomment -%} {%- if url and page.url != url -%} {%- capture wrapped_name -%}{{- wrapped_name -}}{%- endcapture -%} {%- endif -%} {{- wrapped_name -}}
