{%- comment %} The HTML is *VERY* sensitive to new lines, so we fill with {{- null -}} which does nothing _except_ consume a new line on either side. Otherwise it doesn't work inside footnotes. See: https://stackoverflow.com/questions/50467557/jekyll-on-github-pages-how-to-include-external-content-inside-footnotes {% endcomment -%} {%- capture newline %} {% endcapture -%} {%- assign target_name = include.name | replace: newline,' ' | downcase | strip -%} {%- comment -%}Assign link_text here incase we don't find a link. {%- endcomment -%} {%- if include.link_text != '' and include.link_text != null -%} {%- assign link_text = include.link_text | replace: newline,' ' | strip -%} {%- else -%} {%- assign link_text = include.name | replace: newline, ' ' | strip -%} {%- endif -%} {%- assign possessive = include.possessive | default: false -%} {%- comment %}Try to find the book and set a URL{% endcomment -%} {%- assign url = null -%} {%- for page in site.pages -%} {%- assign test_name = page.title | downcase | strip -%} {%- if test_name == target_name and page.layout == "author_page" -%} {%- assign url = page.url -%} {%- comment %}Set the text for the link depending on if we have a link_text or not. Default to the page name so that link text is standardized across articles.{% endcomment -%} {%- if include.link_text != '' and include.link_text != null -%} {%- assign link_text = include.link_text | replace: newline,' ' | strip -%} {%- else -%} {%- assign link_text = page.title | replace: newline, ' ' | strip -%} {%- endif -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- comment %}Wrap the name in tags.{% endcomment -%} {%- capture wrapped_name -%}{{- link_text -}}{%- endcapture -%} {%- comment %}Add 's if possessive{% endcomment -%} {%- if possessive == true -%} {%- capture wrapped_name -%}{{- wrapped_name -}}'s{%- endcapture -%} {%- endif -%} {%- comment %}Wrap again with a link if we find one.{% endcomment -%} {%- if url and page.url != url -%} {%- capture wrapped_name -%}{{- wrapped_name -}}{%- endcapture -%} {%- endif -%} {{- wrapped_name -}}
