{%- comment %} The HTML is *VERY* sensitive to new lines, so we fill with {{- null -}} which does nothing _except_ consume a new line on either side. Otherwise it doesn't work inside footnotes. See: https://stackoverflow.com/questions/50467557/jekyll-on-github-pages-how-to-include-external-content-inside-footnotes {% endcomment -%} {%- capture newline %} {% endcapture -%} {%- assign target_name = include.name | replace: newline,' ' | strip -%} {%- assign url = null -%} {%- assign possessive = include.possessive | default: false -%} {%- assign link_text = include.link_text | default: '' -%} {%- assign link_text = link_text | replace: newline,' ' | strip -%} {%- comment %}Try to find the book and set a URL{% endcomment -%} {%- for page in site.pages -%} {%- if page.author_name == target_name and page.layout == "author_page" -%} {%- assign url = page.url -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- comment %}Wrap the name in tags.{% endcomment -%} {%- capture wrapped_name -%}{{- target_name -}}{%- endcapture -%} {%- comment %}Set {{ target_title }} depending on if we have a link_text or not.{% endcomment -%} {%- if include.link_text != '' and include.link_text != null -%} {%- capture wrapped_name -%}{{- link_text -}}{%- endcapture -%} {%- endif -%} {%- comment %}Add 's if possessive{% endcomment -%} {%- if possessive == true -%} {%- capture wrapped_name -%}{{- wrapped_name -}}'s{%- endcapture -%} {%- endif -%} {%- comment %}Wrap again with a link if we find one.{% endcomment -%} {%- if url and page.url != url -%} {%- capture wrapped_name -%}{{- wrapped_name -}}{%- endcapture -%} {%- endif -%} {{- wrapped_name -}}
