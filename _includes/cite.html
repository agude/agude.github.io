{%- comment -%} A citation formatter that builds the citation string step by step. Non-breaking spaces are only used for specific cases where line breaks would be problematic, like between "vol." and the volume number. {%- endcomment -%} {%- assign citation_string = "" -%} {%- comment -%} Handle author information {%- endcomment -%} {%- if include.author_last and include.author_first -%} {%- assign author_formatted = include.author_last | append: ", " | append: include.author_first | append: ". " -%} {%- assign citation_string = citation_string | append: author_formatted -%} {%- elsif include.author_last -%} {%- assign citation_string = citation_string | append: include.author_last | append: ". " -%} {%- elsif include.author_first -%} {%- assign citation_string = citation_string | append: include.author_first | append: ". " -%} {%- endif -%} {%- comment -%} Add the title with quotation marks {%- endcomment -%} {%- if include.title %} {%- assign title_formatted = '"' | append: include.title | append: '." ' -%} {%- assign citation_string = citation_string | append: title_formatted -%} {%- endif -%} {%- comment -%} Add publication details if available {%- endcomment -%} {%- if include.publication_title -%} {%- assign publication_formatted = '' | append: include.publication_title | append: ', ' -%} {%- assign citation_string = citation_string | append: publication_formatted -%} {%- endif -%} {%- if include.editor -%} {%- assign editor_formatted = 'edited by ' | append: include.editor | append: ', ' -%} {%- assign citation_string = citation_string | append: editor_formatted -%} {%- endif -%} {%- comment -%} Use non-breaking spaces and nowrap class for volume and number abbreviations {%- endcomment -%} {%- if include.volume -%} {%- assign volume_formatted = 'vol. ' | append: include.volume | append: ', ' -%} {%- assign citation_string = citation_string | append: volume_formatted -%} {%- endif -%} {%- if include.number -%} {%- assign number_formatted = 'no. ' | append: include.number | append: ', ' -%} {%- assign citation_string = citation_string | append: number_formatted -%} {%- endif -%} {%- if include.publisher -%} {%- assign publisher_formatted = include.publisher | append: ', ' -%} {%- assign citation_string = citation_string | append: publisher_formatted -%} {%- endif -%} {%- if include.date -%} {%- assign date_formatted = include.date | append: '. ' -%} {%- assign citation_string = citation_string | append: date_formatted -%} {%- endif -%} {%- comment -%} Handle page numbers with non-breaking space and nowrap class {%- endcomment -%} {%- if include.first_page and include.last_page -%} {%- assign pages_formatted = 'pp. ' | append: include.first_page | append: '--' | append: include.last_page | append: '. ' -%} {%- assign citation_string = citation_string | append: pages_formatted -%} {%- elsif include.first_page -%} {%- assign page_formatted = 'p. ' | append: include.first_page | append: '. ' -%} {%- assign citation_string = citation_string | append: page_formatted -%} {%- endif -%} {%- comment -%} Add URL if available {%- endcomment -%} {%- if include.url -%} {%- assign url_formatted = '[' | append: include.url | append: '](' | append: include.url | append: ')' -%} {%- assign citation_string = citation_string | append: url_formatted -%} {%- endif -%} {{ citation_string }}
